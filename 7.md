```
<!DOCTYPE html>
<html>
<head>
    <title>Dynamic Content using DOM</title>

    <script>

        function addContent() {

            // Create heading
            let heading = document.createElement("h2");
            heading.id = "dynamicHeading";
            heading.innerHTML = "Welcome to JavaScript DOM";

            // Create paragraph
            let para = document.createElement("p");
            para.id = "dynamicPara";
            para.innerHTML = "This content is added dynamically using DOM.";

            // Add elements to body
            document.body.appendChild(heading);
            document.body.appendChild(para);
        }

        function clearContent() {

            // Remove heading
            let heading = document.getElementById("dynamicHeading");
            if (heading) {
                heading.remove();
            }

            // Remove paragraph
            let para = document.getElementById("dynamicPara");
            if (para) {
                para.remove();
            }
        }

    </script>
</head>

<body>

    <h1>DOM Example</h1>

    <button onclick="addContent()">
        Add Content
    </button>

    <button onclick="clearContent()">
        Clear Content
    </button>

</body>
</html>
```
