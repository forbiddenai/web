    <h2>PHP String Manipulation Example</h2>

    <?php

        $str = "Hello World";

        echo "<b>Original String:</b> " . $str . "<br><br>";

        // String Length
        echo "<b>String Length:</b> " . strlen($str) . "<br><br>";

        // Reverse String
        echo "<b>Reversed String:</b> " . strrev($str) . "<br><br>";

        // Substring
        echo "<b>Substring:</b> " . substr($str, 0, 5) . "<br><br>";

    ?>
