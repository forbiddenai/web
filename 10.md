<!DOCTYPE html>
<html>
<body>

<form method="post">
  Enter text: <input type="text" name="user_input">
  <input type="submit" value="Process">
</form>

<?php
if ($_SERVER["REQUEST_METHOD"] == "POST" && !empty($_POST['user_input'])) {
    // Get input from the form
    $str = $_POST['user_input'];

    // Process the string
    $len = strlen($str);
    $rev = strrev($str);
    $subs = substr($str, 1, 4);

    // Display results
    echo "<h3>Results:</h3>";
    echo "Original string: " . htmlspecialchars($str) . "<br>";
    echo "Length: " . $len . "<br>";
    echo "Reverse: " . $rev . "<br>";
    echo "Substring (index 1-4): " . $subs . "<br>";
}
?>

</body>
</html>
